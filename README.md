```java
 private static void vocales() {
		char caracter;
		do {
			caracter = Datoschar("Introduce un carácter: ");
			
			switch (caracter) {
			case 'a':
			case 'e':
			case 'i':
			case 'o':
			case 'u':
			System.out.println("Es una vocal");
			break;
			default:
				System.out.println("Es una consonante");
			}
		}while(caracter!= ' ');
  }
````
